\problem{\textit{Rotating loop in a dynamic magnetic field.} If the loop in Fig.Q6.9 uniformly rotates in a low-frequency time-harmonic uniform magnetic field of flux density $B(t) = B_0 \cos \omega t \,(B_0 > 0)$, the induced emf in the loop, with ${\cal E}_0 > 0$, can be represented as 
\begin{enumerate}[label=(\Alph*)]\setlength{\parskip}{1pt}
\item $e_\mathrm{ind}(t) = {\cal E}_0 \cos \omega t$.
\item {$e_\mathrm{ind} (t) = {\cal E}_0 \sin 2\omega t$}.
\item $e_\mathrm{ind}(t) = -{\cal E}_0$.
\item $e_\mathrm{ind}(t) = {\cal E}_0 \sin \omega t$.
\item $e_\mathrm{ind}(t) = 0$.
\end{enumerate}}
