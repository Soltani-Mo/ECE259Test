\problem{\textit{Nonuniformly magnetized ferromagnetic body.} The magnetization vector in a ferromagnetic body is given by $\vect{M} = M(x) \uvect{a}_y$. The magnetization volume current density vector inside this body can be represented as
\begin{enumerate}[label=(\Alph*)]\setlength{\parskip}{1pt}
\item $\vect{J}_m = J_m(x) \uvect{a}_x$.
\item $\vect{J}_m = J_m(x) \uvect{a}_y$.
\item {$\vect{J}_m = J_m(x) \uvect{a}_z$}.
\item $\vect{J}_m = J_m(y) \uvect{a}_x$.
\item $\vect{J}_m = J_m(z) \uvect{a}_x$.
\item $\vect{J}_m = 0$.
\end{enumerate}}

\solution{(C)}
\answer{(C)}
