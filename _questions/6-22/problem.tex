\problem{\textit{Fluid flow through a tube with a static magnetic field.} A liquid of conductivity $\sigma$ flows with a constant velocity $\vect{v} = v \uvect{a}_x$ through a tube of width $d$, in which a uniform time-invariant magnetic field of flux density $\vect{B} = B \uvect{a}_y$ is applied, as depicted in Fig.~Q6.10. The induced electric field intensity vector, $\vect{E}_\mathrm{ind}$, and the field intensity vector due to excess charge, $\vect{E}_q$, in the liquid are given by 
\begin{enumerate}[label=(\Alph*)]\setlength{\parskip}{1pt}
\item $\vect{E}_\mathrm{ind} = v B d \uvect{a}_z$ and $\vect{E}_q = 0$.
\item $\vect{E}_\mathrm{ind} = E_q = vB d \uvect{a}_z$.
\item $\vect{E}_\mathrm{ind} = \vect{E}_q = v B \uvect{a}_z$.
\item $\vect{E}_\mathrm{ind} = vB\uvect{a}_z$ and $\vect{E}_q = 0$.
\item {$\vect{E}_\mathrm{ind} = v B \uvect{a}_z$ and $\vect{E}_q = - v B \uvect{a}_z$}.
\item $\vect{E}_\mathrm{ind} = \vect{E}_q = 0$.
\end{enumerate}
