\problem{\textit{Possible distributions of steady currents. } 
Consider each of the following four vectors in the Cartesian coordinate system: 
$\vect{J}_1 = J_0 \vect{a}_x$, $\vect{J}_2 = J_0 (x / \alpha) \vect{a}_x$, 
$\vect{J}_3 = J_0 (xy / \alpha^2) \vect{a}_x$, $\vect{J}_4 = J_0 (y / \alpha)^2 \vect{a}_x$, 
where $J_0$ and $\alpha$ are constants. Which of them can be the density vector of a steady-state current in a conducting medium? 

\begin{enumerate}[label=(\Alph*)]\setlength{\parskip}{1pt}
\item $\vect{J}_1$ only.
\item $\vect{J}_1$ and $\vect{J}_2$ only. 
\item {$\vect{J}_1$ and $\vect{J}_4$ only.}
\item $\vect{J}_4$ only.
\item All of the vectors. 
\item None of the vectors. 
\end{enumerate}}

\solution{(C)}
\answer{(C)}
