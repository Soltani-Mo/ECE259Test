\problem{\textit{Inductance of a loop with a linear magnetic core.} If a piece of a ferromagnetic material of relative permeability $\mu_r$ is placed as a core of a wire loop, as indicated in Fig.~Q7.3, the inductance of the loop, $L$, is related that, $L_o$, of the same loop with no core as follows:
\begin{center}
\includegraphics[scale = 0.8]{\cpath Q7-7.png}
\end{center}
\begin{enumerate}[label=(\Alph*)]\setlength{\parskip}{1pt}
\item $L = \mu_r L_o$.
\item {$L_o < L < \mu_r L_o$.}
\item $L = L_o$.
\item $L = L_o/ \mu_r$.
\item $L = \mu_r \mu_o L_o$.
\item $L = L_o/(\mu_r \mu_o)$.
\end{enumerate}
($\mu_o$ is the permeability of a vacuum.)}

\solution{(B)}
\answer{(B)}
