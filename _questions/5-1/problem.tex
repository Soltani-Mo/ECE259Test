\problem{\textit{Infinitely long cylindrical bar magnet.} An infinintely long cylindrical bar magnet of radius $a$, is permanently magnetized with a uniform magnetization, and the magnetization vector, of magnitude $M$, is parallel to the bar axis. Magnitudes of the magnetization volume and surface current density vectors, $\vec{J}_m$ and $\vec{J}_{ms}$, over the volume and surface of the magnet, respectively, are
\begin{enumerate}[label=(\Alph*)]\setlength{\parskip}{1pt}
\item $J_m = M$ and $J_{ms} = 0$
\item {$J_m = 0$ and $J_{ms} = M$}
\item $J_m = \mu_0 M$ and $J_{ms} = M/a$
\item $J_m = M/a$ and $J_{ms} = M$
\item $J_m = 0$ and $J_{ms} = 0$
\end{enumerate}
($\mu_0$ is the permeability of vacuum.)}

\solution{(B)}
\answer{(B)}
