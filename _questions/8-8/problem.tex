\problem{\textit{Relationship between conduction and displacement currents.} If the conduction current density at a point of a lossy medium of parameter $\epsilon$, $\mu_o$ and $\sigma$ is given by $J(t) = J_o \sin \omega t$, the displacement current density at that point is of the following form ($J_o$, $\omega$, and $J_{do}$ are positive constant):
\begin{enumerate}[label=(\Alph*)]\setlength{\parskip}{1pt}
\item $J_d(t) = J_{do} \sin \omega t$.
\item $J_d(t) =-J_{do} \sin \omega t$.
\item {$J_d(t) = J_{do} \cos \omega t$}.
\item $J_d(t) = J_{do}$.
\item $J_d(t) = 0$.
\item None of the above.
\end{enumerate}}

\solution{(C)}
\answer{(C)}
