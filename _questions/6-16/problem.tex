\problem{\textit{Electromagnetic induction in a nonlinear magnetic circuit.} A coil of wire, with a low-frequency time-harmonic current $i(t) = I_0 \cos \omega t$, is wound uniformly and densely about a thin toroidal core made of a nonlinear ferromagnetic material that exhibits hysteresis effects. Consider the magnetic flux density, $B(t)$, and magnetic field intensity, $H(t)$, in the core, as well as the induced emf, $e_{ind} (t)$, in the coil. Which of these quantities are time-harmonic functions?
\begin{enumerate}[label=(\Alph*)]\setlength{\parskip}{1pt}
\item $B(t)$ only.
\item {$H(t)$ only}.
\item $B(t)$ and $H(t)$ only.
\item $B(t)$ and $e_\mathrm{ind}(t)$ only.
\item All three quantities.
\item None of the quantities.
\end{enumerate}}

\solution{(B)}
\answer{(B)}
