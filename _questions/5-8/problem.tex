\problem{\textit{Homogenous linear magnetic material.} The conduction current density vector at a point of a homogenous magnetic material of relative permeability $\mu_r$ is $\vec{J}$. The curl of the magnetic flux density vector, $\nabla \times \vec{B}$, at that point equals
\begin{enumerate}[label=(\Alph*)]\setlength{\parskip}{1pt}
\item $\mu_0 \vec{J}$.
\item {$\mu_r \mu_0 \vec{J}$.}
\item $\mu_r \vec{J}$.
\item $(\mu_r - 1) \vec{J}$.
\item $\vec{J}$.
\item $0$.
\end{enumerate}}

\solution{(B)}
\answer{(B)}
