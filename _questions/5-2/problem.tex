\begin{enumerate}[label=(\Alph*)]\setlength{\parskip}{1pt}
\item $\vec{B} = \nabla \times \vec{M}$.
\item $\vec{B} = -\vec{M}$.
\item $\vec{B} = \vec{M} a^2$.
\item {$\vec{B} = \mu_0 \vec{M}$.}
\item $\vec{B} =-\mu_0 \vec{M}$.
\item $\vec{B} = 0$.
\end{enumerate}
Note: there is not external magnetic field.}
