\begin{enumerate}[label=(\Alph*)]\setlength{\parskip}{1pt}
\item {Time-invariant current}.
\item Slowly time-varying current.
\item Rapidly time-varying current.
\item Any time-varying current.
\item Any current.
\item None of the above.
\end{enumerate}}
