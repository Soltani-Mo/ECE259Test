\problem{\textit{Rotating loop in a rotating magnetic field.} Assume that the magnetic flux density vector $\vect{B}$ in Fig.~Q6.9 also rotates, in the same direction as the loop, with an angular velocity $\omega_B$, where $\omega_{B} > \omega$ (vectors $\vect{\omega}$ and $\vect{\omega}_B$ are collinear and in the same direction). With ${\cal E}_{0}$, ${\cal E}_{01}$, and ${\cal E}_{02}$ standing for respective constants, the induced emf in the loop is given by 
\begin{enumerate}[label=(\Alph*)]\setlength{\parskip}{1pt}
\item $e_\mathrm{ind} (t) = {\cal E}_0 \cos \omega t$.
\item $e_\mathrm{ind}(t) = {\cal E}_0 \cos \omega_{B} t$.
\item $e_\mathrm{ind}(t) = {\cal E}_{01} \cos \omega t + {\cal E}_{02} \cos \omega_B t$.
\item {$e_\mathrm{ind} (t) = {\cal E}_0 \cos (\omega_B - \omega) t$}.
\item $e_\mathrm{ind} (t) = {\cal E}_0 \cos (\omega_B + \omega ) t$.
\end{enumerate}}
