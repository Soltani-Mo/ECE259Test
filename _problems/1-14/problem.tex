\problem{A spherical distribution of charge $\rho=\rho_0[1-(R^2/b^2)]$ exists in the region $0\le R \le b$. This charge distribution is concentrically surrounded by a conducting shell with inner radius $R_i$ $(> b)$ and outer radius $R_o$. Determine $\vec{E}$ everywhere.}

\solution{
Due to spherical symmetry of the charge distribution (i.e. no variation of charge density $\rho$ in $\uvect{a}_{\theta}$ or $\uvect{a}_{\phi}$), $\vect{E} = \uvect{a}_R E_R(R)$. Applying Gauss' law with a sphere of radius $R$ as the Gaussian surface.
\begin{enumerate} [label=(\alph*)]
\item for $ 0 \le R \le b$
	\begin{equation*}
		4 \pi R^2 \varepsilon_0 E_R(R) = \int_0^{\pi}\int_0^{2\pi}\int_0^R \rho_0  \left( 1 - \frac{R^2}{b^2} \right) R^2 \sin\theta dR d\phi d\theta\\
	\end{equation*}
	\begin{equation*}
		4 \pi R^2 E_R(R) = \frac{\rho_0}{\varepsilon_0} \int_0^R \left( 1 - \frac{R^2}{b^2} \right) 4 \pi R^2 dR\\
	\end{equation*}
	\begin{equation*}
		E_R(R) = \frac{\rho_0}{\varepsilon_0} R \left(\frac{1}{3} - \frac{R^2}{5 b^2} \right)
	\end{equation*}
\item for $ b \le R \le R_i$
	\begin{equation*}
		4 \pi R^2 E_R(R) = \frac{\rho_0}{\varepsilon_0} \int_0^b \left( 1 - \frac{R^2}{b^2} \right) 4 \pi R^2 dR\\
	\end{equation*}
	\begin{equation*}
		E_R(R) = \frac{2\rho_0 b^3}{15 \varepsilon_0 R^2}
	\end{equation*}
\item for $R_i < R < R_o$
	\begin{equation*}
		E_R(R) = 0
	\end{equation*}
\item for $R > R_o$
	\begin{equation*}
		E_R(R) = \frac{2 \rho_0 b^3}{15 \varepsilon_0 R^2}
	\end{equation*}
\end{enumerate}	
}
\answer{
\begin{enumerate} [label=(\alph*)]
\item for $ 0 \le R \le b$
	\begin{equation*}
		E_R(R) = \frac{\rho_0}{\varepsilon_0} R \left(\frac{1}{3} - \frac{R^2}{5 b^2} \right)
	\end{equation*}
\item for $ b \le R \le R_i$
	\begin{equation*}
		E_R(R) = \frac{2\rho_0 b^3}{15 \varepsilon_0 R^2}
	\end{equation*}
\item for $R_i < R < R_o$
	\begin{equation*}
		E_R(R) = 0
	\end{equation*}
\item for $R > R_o$
	\begin{equation*}
		E_R(R) = \frac{2 \rho_0 b^3}{15 \varepsilon_0 R^2}
	\end{equation*}
\end{enumerate}	
}
