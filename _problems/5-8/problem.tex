\problem{[Cheng P.5-1] Assuming $S$ to be the area of the electrodes in the space-charge-limited vacuum diode in Fig. 5-2, find 
\begin{enumerate}[label=(\alph*)]
\item $V(y)$ and $\vect{E}(y)$ within the interelectrode region,
\item the total amount of charge in the interelectrode region,
\item the total surface charge on the cathode and on the anode,
\item the transit time of an electron from the cathode to the anode with $V_0=200$ (V) and $d=1$ (cm).
\end{enumerate}
\begin{center}
\includegraphics[scale= 0.5]{\ppath Cheng5-1_prompt.png}
\end{center}
}

\solution{
%\begin{center}
%\includegraphics[scale= 0.6]{\ppath 5-1.png}
%\end{center}
\begin{enumerate}[label=(\alph*)]
\item Integrating Eq. (5-16):\begin{align*}
V(y) &= \left[ \frac{qJ}{4\varepsilon_0}\sqrt{\frac{m}{2e}}\right]^{2/3}y^{4/3} = V_0(\frac{y}{d})^\frac{4}{3}\\
\vect{E}(y) &= -\uvect{a}_y \frac{dV(y)}{dy} = -\uvect{a}_y \frac{4}{3}\frac{V_0}{d}(\frac{y}{d})^\frac{1}{3}
\end{align*}
\item \begin{align*}
\rho(y) &= \varepsilon_0 \frac{dE(y)}{dy} = -\frac{4\varepsilon_0 V_0}{qd^2}\left(\frac{y}{d}\right)^{-2/3}\\
Q &= \int_0^d \rho(y)Sdy = -\frac{4\varepsilon_0V_0S}{qd^{4/3}}\int_0^d y^{-2/3}dy = -\frac{4 \varepsilon_{o} V_0}{3d}S.
\end{align*}
\item On the anode, $y=d$, $\rho_s = -\varepsilon_0E(d) = \frac{4\varepsilon_0V_0}{3d}$. Therefore
\begin{align*}
\text{Total surface charge on anode} &= \rho_sS = \frac{4 \varepsilon_{o} V_0 }{3d}S\\
\text{Total charge on cathode} &= 0.
\end{align*}
\item Substituting $V(y) = V_0\left(\frac{y}{d}\right)^{4/3}$ in Eq. (5.12):
\begin{align*}
u &= \frac{dy}{dt} = \left( \frac{2e}{m}V_0 \right)^{1/2} \left(\frac{y}{d}\right)^{2/3}\\
y^{-2/3} dy &= \left(\frac{2eV_0}{md^{4/3}}\right)^{1/2}dt\\
\int y^{-2/3} dy &= \int \left(\frac{2eV_0}{md^{4/3}}\right)^{1/2}dt\\
y &= \left(\frac{2eV_0}{9md^{4/3}}\right)^{3/2}t^3.
\end{align*}
Therefore, the transit time is given by 
\begin{align*}
T_t = 3d\left(\frac{m}{2eV_0}\right)^{1/2}.
\end{align*}
For $V_0=200\,[\text{V}]$, $d=10^{-2}\,[\text{m}]$, $m=9.1\times 10^{-31}\,[\text{kg}]$, and $e=1.6\times 10^{-19}\,[\text{C}]$
\begin{align*}
T_t=3.58\,[\text{ns}]
\end{align*}
\end{enumerate}
}

\answer{
\begin{enumerate}[label=(\alph*)]
\item \begin{align*}
V(y)&=V_0(\frac{y}{d})^\frac{4}{3}\\
\vect{E}(y)&=-\uvect{a}_y \frac{4}{3}\frac{V_0}{d}(\frac{y}{d})^\frac{1}{3}
\end{align*}
\item \begin{align*}
Q &= -\frac{4 \varepsilon_{o} V_0}{3d}S
\end{align*}
\item \begin{align*}
\text{Total charge on anode} &= \frac{4 \varepsilon_{o} V_0 }{3d}S\\
\text{Total charge on cathode} &= 0
\end{align*}
\item \begin{align*}
\text{Transit time}\; T_t=3.58\,ns
\end{align*}
\end{enumerate}}
