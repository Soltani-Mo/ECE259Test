\problem{[Cheng P.3-22] The polarization in a dielectric cube of side $L$ centered at the origin is given by $\vect{P} = P_0(\uvect{a}_xx+\uvect{a}_yy+\uvect{a}_zz)$.
\begin{enumerate}[label=\alph*]
\item Determine the surface and volume bound-charge densities. 
\item Show that the total bound charge is zero. 
\end{enumerate}
}

\solution{
\begin{enumerate}[label=(\alph*)] 
	\item $\rho_{p,s} = P_0 \frac{L}{2}$, on all six faces of the cube, $\rho_{p,v} = -\nabla\cdot \vect{P} = -3P_0$.
	\item $Q_s = (6L^2) \rho_{p,s} = 3 P_0 L^2$.
 $Q_v = (L^3) \rho_{p,v} = -3 P_0 L^3$.
\end{enumerate}}
\answer{
\begin{enumerate}[label=(\alph*)] 
	\item $\rho_{p,s} = P_0 \frac{L}{2}$, on all six faces of the cube, $\rho_{p,v} = -3P_0$.
	\item $Q_s = 3 P_0 L^2$.
 $Q_v = -3 P_0 L^3$.
\end{enumerate}}
