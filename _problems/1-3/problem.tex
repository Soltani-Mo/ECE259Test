\problem{
A cube of edge length $a$ in free space is charged with a volume charge density
\[
\rho_u	= \rho_0 \sin(\pi x/a)\,,\; \:  0 \leq x\leq a\,,
\]
 where $\rho_0$ is a constant. Find the total charge in the cube.}

\solution{
\begin{align*}
Q &=\int_0^a\int_0^a\int_0^a\rho_u\ dxdydz\\
&=\rho_0a^2\int_0^a\sin(\pi x/a)dx\\
&= \frac{2\rho_0 a^3}{\pi}
\end{align*}}
\answer{
\begin{equation*}
Q=\frac{2\rho_0 a^3}{\pi}
\end{equation*}
}
