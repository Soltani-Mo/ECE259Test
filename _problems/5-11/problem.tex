\problem{[Cheng P.5-8] A d-c voltage of 6 (V) applied to the ends of 1 (km) of a conducting wire of 0.5 (mm) radius results in a current of 1/6 (A). Find 
\begin{enumerate}[label=(\alph*)]
\item the conductivity of the sphere,
\item the electric field intensity in the wire,
\item the power dissipated in the wire,
\item the electron drift velocity, assuming electron mobility in the wire to be $1.4\times 10^{-3}$ ($\text{m}^2/\text{V}\cdot\text{s}$).
\end{enumerate}
}

\solution{
\begin{enumerate}[label=(\alph*)]
\item 
\begin{align*}
R=\frac{l}{\sigma S}=\frac{V}{I}\rightarrow \sigma = \frac{lI}{SV} = 3.54 \times 10^7\,[\text{S}/\text{m}].
\end{align*}
\item 
\begin{align*}
E = \frac{V}{l} = 6\times 10^{-3}\,[\text{V}/\text{m}].
\end{align*}
\item 
\begin{align*}
P = VI = 1 \,[\text{W}].
\end{align*}
\item The given electron mobility is that of a good conductor.
\begin{align*}
\rho_e &= -\frac{\sigma}{\mu_e}\\
u &= \mid \frac{J}{\rho_e}\mid = \mid \frac{\mu_e J}{\sigma}\mid = \mid \mu_e E\mid = 8.4\times 10^{-6} [\text{m}/\text{s}].
\end{align*}
\end{enumerate} 
}


\answer{
\begin{enumerate}
\item $R = 3.54 \times 10^7\,[\text{S}/\text{m}]$
\item $E = 6\times 10^{-3}\,[\text{V}/\text{m}]$
\item $P = 1 \,[\text{W}]$
\item $u = 8.4\times 10^{-6} [\text{m}/\text{s}]$
\end{enumerate} }
