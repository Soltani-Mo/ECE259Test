\problem{[Cheng P.5-6] Lightning strikes a lossy dielectric sphere --- $\varepsilon =\varepsilon_0$, $\sigma=10$ (S/m) --- of radius 0.1 (m) at time $t=0$, depositing uniformly in the sphere a total charge 1 (mC). Determine, for all $t$, 
\begin{enumerate}[label=(\alph*)]
\item the electric field intensity both inside and outside the sphere,
\item the current density in the sphere.
\end{enumerate}
}

\solution{
\begin{align*}
\rho_o &= \frac{Q_o}{(4\pi/3)b^3} = \frac{10^{-3}}{(4\pi/3)(0.1)^3}=0.239\,[\text{C}/\text{m}^3]\\
\rho &= \rho_o e^{-(\sigma/\varepsilon)t}
\end{align*}
\begin{enumerate}[label=(\alph*)]
\item 
\begin{align*}
R < b: \vect{E}_i &= \uvect{a}_R \frac{(4\pi/3)R^3\rho}{4\pi\varepsilon R^2}=\uvect{a}_R\frac{\rho_o R}{3\varepsilon}e^{-(\sigma/\varepsilon)t}=\uvect{a}_R \,7.5\times10^9 R \,e^{-9.42\times 10^{11}t}\,[\text{V}/\text{m}]\\
R > b: \vect{E}_o &= \uvect{a}_R \frac{Q_o}{4\pi\varepsilon_0 R^2}=\uvect{a}_R \frac{9}{R^2} \times 10^6\,[\text{V}/\text{m}]
\end{align*}
\item 
\begin{align*}
R < b: \vect{J}_i &= \sigma \vect{E}_i =\uvect{a}_R\, 7.5\times10^{10} R\, e^{-9.42\times 10^{11}t}\,[\text{A}/\text{m}^2]\\
R > b: \vect{J}_o &= 0\,[\text{A}/\text{m}^2]
\end{align*}
\end{enumerate}
}

\answer{
\begin{enumerate}[label=(\alph*)]
\item 
\begin{align*}
R < b: \vect{E}_i &=\uvect{a}_R \,7.5\times10^9 R \,e^{-9.42\times 10^{11}t}\,[\text{V}/\text{m}]\\
R > b: \vect{E}_o &=\uvect{a}_R \frac{9}{R^2} \times 10^6\,[\text{V}/\text{m}]
\end{align*}
\item 
\begin{align*}
R < b: \vect{J}_i &= \uvect{a}_R\, 7.5\times10^{10} R\, e^{-9.42\times 10^{11}t}\,[\text{A}/\text{m}^2]\\
R > b: \vect{J}_o &= 0\,[\text{A}/\text{m}^2]
\end{align*}
\end{enumerate}}
