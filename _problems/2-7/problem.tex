\problem{[Cheng P.3-13] Determine the work done in carrying a $-2$ ($\mu$C) charge from $P_1(2,1,-1)$ to $P_2(8,2,-1)$ in the field $\vect{E} = \uvect{a}_xy+\uvect{a}_yx$
\begin{enumerate}[label=(\alph*)]
\item along the parabola $x=2y^2$,
\item along straight line joining $P_1$ and $P_2$.
\end{enumerate}
}

\solution{
For both paths we will use the formula 
\begin{align*}
W = -q\int \vect{E}\cdot d\vect{l}
W = -q\int x\textrm{d}x+y\textrm{d}y.
\end{align*}
\begin{enumerate}[label=(\alph*)]
\item Along the parabola $x=2y^2$: 
\begin{align*}
\textrm{d}x&=4y\textrm{d}y\\
W&=-q\int_{1}^{2}6y^2\textrm{d}y\\
W &= 28\,\text{\textmu J}.
\end{align*}
\item Along straight line $x=6y-4$: 
\begin{align*}
\textrm{d}x&=6\textrm{d}y\\
W&=-q\int_{1}^{2}12y-4\textrm{d}y\\
W&= 28\,\text{\textmu J}.
\end{align*}
\end{enumerate}
As we can see, this confirms that the static electric field is conservative.}

\answer{
\begin{enumerate}[label=(\alph*)]
		\item Along the parabola: $W = 28\,\text{\textmu J}$.
		\item Along straight line: $W= 28\,\text{\textmu J}$.
	\end{enumerate}}
