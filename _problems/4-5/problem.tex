\problem{[Cheng P.3-43] Prove that Eqs (3-180) for stored electrostatic energy hold true for any two-conductor capacitor.}

\solution{
Two conductors at potentials $V_1$ and $V_2$ carrying charges $Q$ and $-Q$:

\begin{align*}
W_e &= \frac{1}{2} V_1 \int_{S_1} \rho_{si}\,dS + \frac{1}{2} V_2 \int_{S_2} \rho_{s2}\,dS 
= \frac{1}{2} Q (V_1  - V_2) 
= \frac{1}{2} CV^2  \,,\quad \quad V = V_1 - V2\,.
\end{align*}}
\answer{
\begin{align*}
W_e &=\frac{1}{2} Q (V_1  - V_2)
\end{align*}}
