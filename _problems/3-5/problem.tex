\problem{[Cheng P.3-26] Determine the boundary conditions for the tangential and normal components of $\vect{P}$ at an interface between two perfect dielectric media with dielectric constants $\varepsilon_{r1}$ and $\varepsilon_{r2}$. }

\solution{
We know that 
\begin{align*}
\vect{P}_{t/n} &= \varepsilon_0 (1-\varepsilon_r)\vect{E}_{t/n},
\end{align*}
for both the tangential and normal components. Therefore, we simply use this relation with the boundary conditions that we already know for the tangential and normal components of $\vect{E}$. This yields
\begin{align*}
\vect{E}_{t1} &= \vect{E}_{t2}\\
\frac{1}{\varepsilon_{r1} - 1} \vect{P}_{t1} &= \frac{1}{\varepsilon_{r2} - 1} \vect{P}_{t2} \\[0.25em]
\varepsilon_{r1} \vect{E}_{n1} &= \varepsilon_{r2}\vect{E}_{n2}\\
\frac{\varepsilon_{r1}}{\varepsilon_{r1} - 1} \vect{P}_{n1} &= \frac{\varepsilon_{r2}}{\varepsilon_{r2} - 1} \vect{P}_{n2} \,.
\end{align*}}
\answer{
\begin{align*}
\frac{1}{\varepsilon_{r1} - 1} \vect{P}_{t1} &= \frac{1}{\varepsilon_{r2} - 1} \vect{P}_{t2} \\[0.25em]
\frac{\varepsilon_{r1}}{\varepsilon_{r1} - 1} \vect{P}_{n1} &= \frac{\varepsilon_{r2}}{\varepsilon_{r2} - 1} \vect{P}_{n2} \,.
\end{align*}}
