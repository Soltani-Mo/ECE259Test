\problem{
Three point charges $Q=-1$\,nC are placed at three vertices $(a,0,0)$, $(0,a,0)$ and $(0,0,a)$ of a cube with $a=1$\,m. Find the electric field intensity vector at (a) the coordinate origin $(0,0,0)$ and (b) the point on the $z$-axis $(0,0,100\,\text{m})$.}

\solution{
\begin{enumerate}[label=(\alph*)]
	\item $\vect{E}=\dfrac{1\times10^{-9}}{4\pi\varepsilon_0}(\uvect{a}_x+\uvect{a}_y+\uvect{a}_z)=9(\uvect{a}_x+\uvect{a}_y+\uvect{a}_z)\,\text{N}/\text{C}=9\sqrt{3}\uvect{a}_r\,\text{N}/\text{C}$,\\ where $\uvect{a}_r=\dfrac{1}{\sqrt{3}}(\uvect{a}_x+\uvect{a}_y+\uvect{a}_z)$
	\item From the large distance (100\m), the three charges can be treated as a single charge $Q_\mathrm{tot}=-3\nC$, leading to:
	\begin{equation*}
		\vect{E}=\dfrac{3\times 10^{-9}}{4\pi\varepsilon_0 100^2}=2.7\,\uvect{a}_z\,\text{mN}/\text{C} 
	\end{equation*}
\end{enumerate}}

\answer{
\begin{enumerate}[label=(\alph*)]
\item $\vect{E}=9\sqrt{3}\uvect{a}_r\,\text{N}/\text{C}$,\\ where $\uvect{a}_r=\dfrac{1}{\sqrt{3}}(\uvect{a}_x+\uvect{a}_y+\uvect{a}_z)$
\item $\vect{E}=2.7\,\uvect{a}_z\,\text{mN}/\text{C}$
\end{enumerate}}
