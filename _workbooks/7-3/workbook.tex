\textbf{Goal:} Determine the magnetic flux density inside an infinitely long solenoid in free space having $n$ closely wounds turns per unit length and carrying a current $I$.
\begin{center}
	(a) \includegraphics[width=0.5\textwidth]{\wpath Problem2_Figure1.png}\\
	(b) \includegraphics[width=0.7\textwidth]{\wpath Problem2_Figure2.png} 
\end{center}
\textbf{Steps:} 
\begin{enumerate}

\item State Amp\`{e}re's law.\\
%\vskip 50pt
\solution{
The integral form of the Amp\`{e}re's circuit law is
\begin{equation*}
\oint_C \vec{B} \cdot d\vec{l} = \mu_0 I_\text{enc}
\end{equation*}
where $C$ is the contour of the boundary points of a surface $S$ and $I_\text{enc}$ is the total current passing through surface $S$.
}

\item Sketch the magnetic field inside the solenoid in diagram (b) above.\\

\item Applying Amp\`{e}re's law, determine the magnetic field inside the solenoid.\\
%\vskip 100pt
\solution{
\begin{eqnarray*}
BL &=& \mu_0 n L I\\
B &=& \mu_0 n I
\end{eqnarray*}
}

\end{enumerate}

\answer{\begin{eqnarray*}
B &=& \mu_0 n I
\end{eqnarray*}}
