%Cheng 6-38
\textbf{Goal:} Determine the mutual inductance between a very long, straight wire and a conducting equilateral triangle loop, as shown in the figure below.
\begin{center}
\centering
\includegraphics[width=0.2\textwidth]{\wpath Cheng6-38.png}
\end{center}
\textbf{Steps:} 
\begin{enumerate}
\item Define the mutual inductance $L_{12}$.
%\vskip 100pt

\solution{
\begin{equation*}
L_{12} = \frac{\Phi_{12}}{i_1}
\end{equation*}
where $\Phi_{12}$ is the total magnetic flux through the second loop due to the $B$-field generated by $i_1$ in the first loop.}

\item What is the $B$-field generated by the line current?
%\vskip 100pt

\solution{
\begin{equation*}
\vec{B} = \vec{a}_\phi \frac{\mu_o I}{2\pi r}
\end{equation*}}

\item Find the magnetic flux through the triangle due to the $B$-field generated by the line current.
%\vskip 200pt

\solution{
\begin{eqnarray*}
\Phi_{12} &=& \int_S \vec{B} \cdot d\vec{s} = \int_S B_\phi \frac{2}{\sqrt{3}} (r-d)\,dr = \frac{\mu_o I}{\pi \sqrt{3}} \int_d^{d+\frac{\sqrt{3}}{2}b} \left( \frac{r - d}{r} \right)\,dr\\
&=& \frac{\mu_o I}{\pi \sqrt{3}} \left[ \frac{\sqrt{3}}{2} b - d \ln \left(1 + \frac{\sqrt{3}b}{2d}\right) \right]
\end{eqnarray*}}

\item What is the mutual inductance?
%\vskip 100pt

\solution{
\begin{equation*}
L_{12} = \frac{\mu_o}{\pi \sqrt{3}} \left[ \frac{\sqrt{3}}{2} b - d \ln \left(1 + \frac{\sqrt{3}b}{2d}\right) \right]
\end{equation*}}

\end{enumerate}
\answer{
\begin{equation*}
L_{12} = \frac{\mu_o}{\pi \sqrt{3}} \left[ \frac{\sqrt{3}}{2} b - d \ln \left(1 + \frac{\sqrt{3}b}{2d}\right) \right]
\end{equation*}
}
